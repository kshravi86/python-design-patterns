class Node(object):
    def __init__(self,data,next = None):
        self.data = data
        self.next = next

class linklist(object):
    
    def __init__(self):
        self.head = 0

    
    def Initiallist(self,data):
        if len(data) == 1:
            self.head = Node(data[0])
        else:
            self.head = Node(data[0])
            p = self.head
            for i in data[1:]:
                p.next = Node(i)
                p = p.next
        return

        return self.head == 0

            print "Linklist Is Empty!"

    def Lengthlist(self):
        length = 0
        p = self.head

            p = p.next
            length = length + 1
        return length

    
    def Clearlist(self):
        self.head = 0
